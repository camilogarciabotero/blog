Data manipulation

all_artists <- artists |>
  mutate(across(state:type, as_factor)) |>
  drop_na(artists_n) |>
  group_by(race) |>
  summarize(n = sum(artists_n))
all_artists

Data visualization

ggplot(all_artists) +
  aes(y = reorder(race, n), x = n, fill = race, label = n) +
  geom_col() +
  geom_text() +
  theme_bw() +
  theme(
    legend.position = "none",
    axis.title.y = element_blank()
  ) +
  labs(
    title = "Number of artists across USA according to race"
  )

Alluvial plots

factored_artists <- artists |>
  mutate(across(state:type, as_factor)) |> 
  group_by(race, type, state) |> 
  summarise(artists_n) |>
  drop_na()
`summarise()` has grouped output by 'race', 'type'. You can override using the
`.groups` argument.
library(ggalluvial)
library(ggfittext)

ggplot(factored_artists) +
  aes(y = artists_n, axis1 = race, axis2 = type, fill = race) +
  geom_stratum(alpha = .5) +
  geom_alluvium() +
  geom_fit_text(stat = "stratum", aes(label = after_stat(stratum))) +
  theme_bw() +
  theme(
    legend.position = "none"
  ) +
  scale_fill_viridis_d() +
  labs(
    y = ""
  )
